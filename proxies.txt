# Формат: protocol://[user:pass@]host:port
# Рекомендуется SOCKS5 — работает для всех векторов включая TCP-уровень

# SOCKS5 без авторизации
socks5://1.2.3.4:1080
socks5://5.6.7.8:1080
socks5://9.10.11.12:1080
socks5://13.14.15.16:1080
socks5://17.18.19.20:1080
socks5://21.22.23.24:1080
socks5://25.26.27.28:1080

# SOCKS5 с авторизацией
# socks5://user:password@1.2.3.4:1080

# HTTP прокси (только для HTTP/HTTPS векторов, НЕ для slowloris/ssl)
# http://1.2.3.4:8080
# http://user:password@1.2.3.4:3128
